  <h3>Output:</h3>

  <script>
    function parseGraph(nodesTxt, edgesTxt) {
      const nodesLines = nodesTxt.trim().split("\n").slice(1);
      const nodes = nodesLines.map(l => {
        const [label, plates] = l.split("|");
        return { label, plates };
      });

      const edgesLines = edgesTxt.trim().split("\n").slice(1);
      const edges = edgesLines.map(l => {
        const [from,to] = l.split(",");
        return { from, to };
      });

      return { nodes, edges };
    }

    document.getElementById("btn-parse").addEventListener("click", ()=>{
      const nodesTxt = document.getElementById("nodesTxt").value;
      const edgesTxt = document.getElementById("edgesTxt").value;
      const parsed = parseGraph(nodesTxt, edgesTxt);
      document.getElementById("output").textContent = JSON.stringify(parsed, null, 2);
    });
  </script>
