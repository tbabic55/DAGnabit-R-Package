<body>
  <h1>DAGnabitWidget Demo</h1>
  <div id="widgetContainer"></div>
  <div class="controls">
    <button id="btn-simple">Simple DAG</button>
    <button id="btn-chain">Chain DAG</button>
    <button id="btn-clear">Clear</button>
  </div>

  <script>
    (function(){
      const container = document.getElementById("widgetContainer");

      function renderWidget(data) {
        container.innerHTML = "";
        if (typeof window.HTMLWidgets !== "undefined") {
          HTMLWidgets.widgets.DAGnabitWidget.factory(container, 600, 400).renderValue(data);
        } else {
          container.innerHTML = "<p style='color:red;'>Error: DAGnabitWidget not loaded</p>";
        }
      }

      const SIMPLE = {
        nodes: [{id:1,label:"A"}, {id:2,label:"B"}],
        edges: [{from:1,to:2}]
      };
      const CHAIN = {
        nodes: [{id:1,label:"A"},{id:2,label:"B"},{id:3,label:"C"}],
        edges: [{from:1,to:2},{from:2,to:3}]
      };

      document.getElementById("btn-simple").onclick = () => renderWidget(SIMPLE);
      document.getElementById("btn-chain").onclick = () => renderWidget(CHAIN);
      document.getElementById("btn-clear").onclick = () => container.innerHTML = "";

      renderWidget(SIMPLE);
    })();
  </script>
</body>
